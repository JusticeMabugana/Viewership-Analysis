 Aggregate total time watched by platform
platform_totals = df.groupby('Platform')['TotalTimeWatched'].sum()

# Plot the aggregated data
plt.bar(
    platform_totals.index,
    platform_totals.values,
    color='skyblue'
)
plt.xlabel('Platform')
plt.ylabel('Total Time Watched')
plt.title('Total Time Watched per Platform')
plt.xticks(rotation=45)
plt.tight_layout()
plt.show()
